document.addEventListener('DOMContentLoaded', () => { const commandInput = document.getElementById('command-input'); const output = document.getElementById('output'); const prompt = '$ '; // Frames for ASCII art heart animation const heartFrames = [ ` ** ** ** * * ** ** * ** ** ** ** ** * `, ` ** ** ** * ** ** * * ** ** * ** ** ** ** ** * `, ` ** ** ** * ** ** * * ** ** * ** ** ** ** ** * `, ` ** ** ** * * ** ** * ** ** ** ** ** * `, ]; let currentFrame = 0; let animationInterval; function startHeartAnimation() { if (animationInterval) { clearInterval(animationInterval); } animationInterval = setInterval(() => { const frame = heartFrames[currentFrame]; output.innerHTML += `
${frame}
`; currentFrame = (currentFrame + 1) % heartFrames.length; output.scrollTop = output.scrollHeight; }, 500); // Change frame every 500 milliseconds } function handleCommand(command) { switch (command.trim()) { case 'help': return 'Available commands: help, echo [text], clear, heart'; case 'clear': output.innerHTML = ''; return ''; case 'heart': startHeartAnimation(); return 'Heart animation started. Use "clear" to stop it.'; default: if (command.startsWith('echo ')) { return command.slice(5); } return `Command not found: ${command}`; } } function executeCommand() { const command = commandInput.value; const response = handleCommand(command); // Display command and response output.innerHTML += `
${prompt}${command}
`; if (response) { output.innerHTML += `
${response}
`; } // Clear the input commandInput.value = ''; // Scroll to the bottom output.scrollTop = output.scrollHeight; } commandInput.addEventListener('keydown', (event) => { if (event.key === 'Enter') { executeCommand(); event.preventDefault(); // Prevents form submission or other default actions } }); });
